## Install

```bash
npm install --save create-react-hook-sample
```

## Usage

```tsx
import * as React from 'react'

import { useMyHook } from 'create-react-hook-sample'

const Example = () => {
  const example = useMyHook()
  return (
    <div>
      {example}
    </div>
  )
}
```
